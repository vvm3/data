// Zod imports
import { messages } from "@/config/messages"
import { z } from "zod"

// Dynamic Zod Schemas
export const schemeFormSchema = z.object({
  id: z.string().nullable(),
  schemeName: z.string().min(1, messages.SchemeAndMarks.schemeValidationError),
  credits: z.string().refine(val => !isNaN(Number(val)), { message: messages.SchemeAndMarks.creditValidationError}),
  pedagogy: z.record(z.coerce.number().min(0)),
  marks: z.record(z.object({
    internal: z.coerce.number().min(0),
    final: z.coerce.number().min(0)
  }))
})

//type
export type DynamicSchemeFormData = z.infer<typeof schemeFormSchema>
